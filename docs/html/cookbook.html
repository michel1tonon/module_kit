Cookbook

Short and practical recipes.

Recipe 1: go_router + provider

final composer = ModuleComposer<RouteBase, SingleChildWidget>()
  ..addAll([FeatureA(), FeatureB()]);

return ModuleComposerBuilder<RouteBase, SingleChildWidget>(
  composer: composer,
  builder: (context, {required injectors, required routers}) {
    return MultiProvider(
      providers: injectors,
      child: MaterialApp.router(
        routerConfig: GoRouter(routes: routers),
      ),
    );
  },
);

Recipe 2: Navigator + get_it

typedef RouteEntry = MapEntry<String, WidgetBuilder>;
typedef RegisterDependency = void Function(GetIt sl);

final composer = ModuleComposer<RouteEntry, RegisterDependency>();
final enabled = await composer.getAllEnabledModules(context);
final routes = composer.getRouters(context, enabled);
final regs = composer.getInjectors(context, enabled);

Recipe 3: go_router + riverpod overrides

final composer = ModuleComposer<RouteBase, Override>();

return ModuleComposerBuilder<RouteBase, Override>(
  composer: composer,
  builder: (context, {required injectors, required routers}) {
    return ProviderScope(
      overrides: injectors,
      child: MaterialApp.router(
        routerConfig: GoRouter(routes: routers),
      ),
    );
  },
);
